\documentclass[12pt, a4paper]{article}

%% ===== Packages =====
\usepackage{geometry}
\geometry{margin=2.5cm}
\usepackage{fontspec}
\usepackage{xeCJK}
\usepackage{hyperref}
\usepackage{fancyhdr}
\usepackage{titlesec}
\usepackage{xcolor}
\usepackage{listings}
\usepackage{longtable}
\usepackage{booktabs}
\usepackage{array}
\usepackage{setspace}

%% ===== Chinese Font Configuration =====
%% CJK fonts — set via pandoc variables or defaults here
%% 正文: 宋体(SimSun), 标题: 黑体(SimHei)
$if(CJKmainfont)$
\setCJKmainfont{$CJKmainfont$}
$else$
\setCJKmainfont{SimSun}
$endif$

$if(CJKsansfont)$
\setCJKsansfont{$CJKsansfont$}
$else$
\setCJKsansfont{SimHei}
$endif$

%% Define \heiti and \songti commands for heading formatting
\newcommand{\heiti}{\CJKfamily{sf}}
\newcommand{\songti}{\CJKfamily{rm}}

%% Chinese font sizes (字号)
\newcommand{\zihao}[1]{%
  \ifnum 1=#1 \fontsize{26pt}{36pt}\selectfont\fi
  \ifnum 2=#1 \fontsize{22pt}{28pt}\selectfont\fi
  \ifnum 3=#1 \fontsize{16pt}{24pt}\selectfont\fi
  \ifnum 4=#1 \fontsize{14pt}{21pt}\selectfont\fi
  \ifnum -4=#1 \fontsize{12pt}{18pt}\selectfont\fi
  \ifnum 5=#1 \fontsize{10.5pt}{15.75pt}\selectfont\fi
  \ifnum -5=#1 \fontsize{9pt}{13.5pt}\selectfont\fi
}

%% Latin fonts
$if(mainfont)$
\setmainfont{$mainfont$}
$else$
\setmainfont{Times New Roman}
$endif$

$if(monofont)$
\setmonofont{$monofont$}
$else$
\setmonofont{Menlo}
$endif$

%% ===== Colors =====
\definecolor{linkcolor}{RGB}{0, 102, 204}
\definecolor{codebackground}{RGB}{248, 248, 248}
\definecolor{headingcolor}{RGB}{51, 51, 51}

%% ===== Hyperlinks =====
\hypersetup{
    colorlinks=$if(colorlinks)$$colorlinks$$else$true$endif$,
    linkcolor=$if(linkcolor)$$linkcolor$$else$linkcolor$endif$,
    urlcolor=$if(urlcolor)$$urlcolor$$else$linkcolor$endif$,
    citecolor=$if(citecolor)$$citecolor$$else$linkcolor$endif$,
    pdfauthor={$author$},
    pdftitle={$title$}
}

%% ===== Page Style =====
%% Margins: A4 standard (上下2.54cm, 左右3.17cm)
\geometry{top=2.54cm, bottom=2.54cm, left=3.17cm, right=3.17cm}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small\leftmark}
\fancyhead[R]{\small\thepage}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0pt}

%% ===== Line Spacing =====
$if(linestretch)$
\setstretch{$linestretch$}
$else$
\setstretch{1.5}
$endif$

%% ===== Section Formatting =====
%% 一级标题: 黑体三号(16pt), 段前段后0.5行
\titleformat{\section}
    {\zihao{3}\heiti\color{headingcolor}}
    {\thesection}{1em}{}
\titlespacing{\section}{0pt}{0.5\baselineskip}{0.5\baselineskip}

%% 二级标题: 黑体四号(14pt)
\titleformat{\subsection}
    {\zihao{4}\heiti\color{headingcolor}}
    {\thesubsection}{1em}{}

%% 三级标题: 宋体小四(12pt)加粗
\titleformat{\subsubsection}
    {\zihao{-4}\songti\bfseries}
    {\thesubsubsection}{1em}{}

%% ===== First-line indent (首行缩进2字符) =====
\usepackage{indentfirst}
\setlength{\parindent}{2em}

%% ===== Code Blocks =====
\lstset{
    backgroundcolor=\color{codebackground},
    basicstyle=\ttfamily\small,
    breaklines=true,
    frame=single,
    numbers=left,
    numberstyle=\tiny\color{gray}
}

%% ===== Title Block =====
$if(title)$
\title{$title$}
$endif$
$if(author)$
\author{$author$}
$endif$
$if(date)$
\date{$date$}
$else$
\date{}
$endif$

\begin{document}

$if(title)$
\maketitle
$endif$

\tableofcontents

$endif$

$body$

\end{document}
